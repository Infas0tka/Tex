\begin{equation*}
\quad \fbox{$\approx$}
 \begin{cases}
   \sum\limits_{k=1}^n A_k \, \rho(x_k) \, f(x_k),
   \\
   \sum\limits_{k=1}^n B_k \, f(x_k),
 \end{cases}
\end{equation*}
